let money = prompt("Ваш бюджет на месяц?", ''),
time = prompt('Введите дату в формате YYYY-MM-DD', '');

let appData = {

Создание в файле скрипта 2 переменных (money и time), которые будут получать данные от пользователя.

Чтобы создать переменные используем функцию let, а именно:

let money=promt("Ваш бюджет на месяц?", "");
time= promt("Введите дату в формате YYYY-MM-DD", "");

Что мы сделали, мы создали две переменных money и time и указали, что эти переменные содержат клманду с вводом ответа от пользователя, на заданные в них вопросы.

Создаем объект appData и помещаем в него объекты

В данном задании необходимо создать массив, который будет содержать несколько данных.

Создаем объект appData со свойствами:

let appData= {
buget: money,
timeData: time,
expenses: {},
optionalExpenses:{},
income: [],
saving: false
};

Мы создали объект appData в которой с помощью фигурных скобок и перечислили входящие в него свойства, а именно:

  • buget - в которое помещаются данные введенные пользователем в переменную money;
  • timeData - в которое помещаются данные введенные пользователем в переменную time;
  • expenses - которое становиться объектом содержащим данные пользователя с обязательными расходами(их мы пока оставим пустуми);
  • optionalExpenses - которое становиться объектом содержащим данные пользователя с необязательными расходами(он также пустой);
  • income - которое будет содержать массив данных о дополнительных доходах(он также пока пустой);
  • saving - имеющее значение - false
Немного дополнительной информации о которой ранее умолчал автор.

Наиболее важная часть объекта – его свойства.
Объект представляет из себя набор свойств, и каждое свойство состоит из имени и значения, ассоциированного с этим именем.
Именем свойства (ключом) может быть только строка, а вот значением – любой тип данных.
Если значением свойства является функция, то её можно назвать методом объекта.

Свойства объектов в JavaScript являются теми же самыми переменными, за тем исключением, что они закреплены за объектом.
Как и все переменные JavaScript, имя объекта (которое тоже может быть переменной) и имя свойства чуствительны к регистру.
Свойства объекта определяют его индивидуальные характеристики.

Получить значение свойства можно с помощью операторов точки (.) или квадратных скобок ([]).
Значение в левой части оператора должно быть ссылкой на объект, обычно это просто имя переменной, содержащей ссылку на объект, но это может быть любое допустимое в JavaScript выражение, являющееся объектом.

Значение в правой части оператора должно быть именем свойства.
При использовании оператора точка (.) справа должен находиться простой идентификатор, соответствующий имени свойства.
Доступ к любому имени свойства, которое содержит невалидный JavaScript идентификатор (например, имя свойства содержит в себе пробел, тире или начинается с цифры), может быть получен с использованием квадратных скобок ([]).

Например:

var person = {
name: "Вася",
"year of birth": 1980,
specialty: "юрист-консультант"
}
alert( person.name ); // Вася
alert( person['name'] ); // Вася
alert( person.specialty ); // юрист-консультант
// 'year of birth' – выражение
alert( person['year of birth'] ); // 1980

Значение свойства может быть любым: как значением простого типа, так и другим объектом.
Значения можно вкладывать друг в друга, причем вложений можно делать сколько угодно:

var toy = {
say: {
ask: {
yes: 'Yes!',
no: 'No!',
},
go: {
toLeft: 'Влево',
toRight: 'Вправо'
},
},
};

document.write( toy.say.ask.yes + '
' ); // Yes!
document.write( toy.say.ask.no + '
' ); // No!
document.write( toy.say.go.toRight ); // Вправо

Значение объекта может быть обновлено путем присваивания.
Если свойство с таким именем уже существует в объекте, его значение заменяется:

var person = {
name: "Вася",
"year of birth": 1980,
specialty: "юрист-консультант"
}
person.fullName = "Василий Иванов"; // Создать свойство "fullName"
person['year of birth'] = 1985; // Заменить свойство
alert( person.fullName ); // Василий Иванов
alert( person['year of birth'] ); // 1985

Квадратные скобки также позволяют обратиться к свойству, имя которого хранится в переменной:

var person = {};
person.name = "Вася";
var key = 'name';

alert( person[key] ); // Вася
alert( person.key ); // undefined

Если имя свойства хранится в переменной, то единственный способ к нему обратиться – это применение оператора [] (квадратные скобки).

Задать пользователю по 2 раза вопросы

В этом задании нам необходимо вновь создать две переменные с вопросом к пользователю, причем вопросы, указанные в переменных должны быть заданы пользователю 2 раза.

Создадим переменные с вопросом:

let a1=promt("Введите обязательную статью расходов в этом месяце", ""),
a2 = prompt("Во сколько обойдется?", ''),
a3 = prompt("Введите обязательную статью расходов в этом месяце", ''),
a4 = prompt("Во сколько обойдется?", '');

Теперь нам надо записать ответы в объект expenses в формате:
expenses: {
“ответ на первый вопрос” : “ответ на второй вопрос”
}

Пишем следующий код:
appData.expenses.a1 = a2;
appData.expenses.a3 = a4;

В данном случаи мы написали, что в объект appData в его свойство expenses внести следующее:
переменную a1, которой присвоить в качестве значения переменную a2
переменную a3, которой присвоить в качестве значения переменну a4
